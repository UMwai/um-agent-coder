### 1. `command/autopilot/vol_regime.py`
The core multi-signal classifier. It computes scores across various volatility metrics, tracks regime transitions, and generates dynamic parameter overrides.

```python
from __future__ import annotations

import logging
import numpy as np
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Dict, List, Optional

from command.autopilot.market_monitor import MarketSnapshot, VolatilitySnapshot, OptionsSnapshot

log = logging.getLogger(__name__)


@dataclass
class RegimeParameters:
    """Strategy parameter overrides generated by the current volatility regime."""
    # DecisionEngine overrides
    regime_scale: float
    min_confidence: float
    max_open_decisions: int
    
    # DeltaNeutralStrategy overrides
    iv_hv_zscore_threshold: float
    max_position_pct: float
    max_total_exposure_pct: float
    
    # UnifiedRiskEngine overrides
    max_daily_loss_pct: float
    warning_drawdown_pct: float


@dataclass
class TransitionEvent:
    """Represents a confirmed shift in market volatility regime."""
    from_regime: str
    to_regime: str
    confidence: float
    trigger_signals: List[str] = field(default_factory=list)
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


@dataclass
class RegimeResult:
    """The complete output of the VolRegimeClassifier for a given snapshot."""
    regime: str
    confidence: float
    sub_scores: Dict[str, float] = field(default_factory=dict)
    parameters: Optional[RegimeParameters] = None
    transition_event: Optional[TransitionEvent] = None
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


class VolRegimeClassifier:
    """
    Multi-signal Volatility Regime Classifier.
    Evaluates market snapshots to classify the environment into one of 5 regimes:
    low_vol, normal, elevated, high_vol, crisis.
    """
    
    REGIMES = ["low_vol", "normal", "elevated", "high_vol", "crisis"]
    
    def __init__(self, history_window: int = 20, confirmation_count: int = 2):
        self.history_window = history_window
        self.confirmation_count = confirmation_count
        
        self.confirmed_regime: str = "normal"
        self.raw_history: List[str] = []
        self.result_history: List[RegimeResult] = []
        
    def classify(self, snap: MarketSnapshot) -> RegimeResult:
        """Evaluates a MarketSnapshot and returns the classified RegimeResult."""
        sub_scores = self._calculate_sub_scores(snap)
        
        # Aggregate score maps to 0-100
        total_score = float(np.mean(list(sub_scores.values())))
        raw_regime = self._score_to_regime(total_score)
        
        self.raw_history.append(raw_regime)
        if len(self.raw_history) > self.history_window:
            self.raw_history.pop(0)
            
        transition_event = None
        
        # Require 'confirmation_count' consecutive snapshots to confirm a transition
        if len(self.raw_history) >= self.confirmation_count:
            recent_raw = self.raw_history[-self.confirmation_count:]
            if all(r == raw_regime for r in recent_raw) and raw_regime != self.confirmed_regime:
                transition_event = TransitionEvent(
                    from_regime=self.confirmed_regime,
                    to_regime=raw_regime,
                    confidence=total_score / 100.0 if raw_regime in ["high_vol", "crisis"] else 1.0 - (total_score / 100.0),
                    trigger_signals=[k for k, v in sub_scores.items() if v > 60] if raw_regime in ["high_vol", "crisis"] else [],
                    timestamp=snap.timestamp
                )
                self.confirmed_regime = raw_regime
                
        params = self._generate_parameters(self.confirmed_regime)
        
        result = RegimeResult(
            regime=self.confirmed_regime,
            confidence=1.0 if self.confirmed_regime == raw_regime else 0.5,
            sub_scores=sub_scores,
            parameters=params,
            transition_event=transition_event,
            timestamp=snap.timestamp
        )
        
        self.result_history.append(result)
        if len(self.result_history) > self.history_window:
            self.result_history.pop(0)
            
        return result
        
    def _calculate_sub_scores(self, snap: MarketSnapshot) -> Dict[str, float]:
        scores = {}
        vol = snap.volatility
        
        # 1. VIX Level
        if vol.vix < 15: scores["vix_level"] = 10.0
        elif vol.vix < 20: scores["vix_level"] = 30.0
        elif vol.vix < 25: scores["vix_level"] = 50.0
        elif vol.vix < 35: scores["vix_level"] = 75.0
        else: scores["vix_level"] = 100.0
            
        # 2. VIX Velocity (Rate of change)
        if vol.vix_change <= 0: scores["vix_velocity"] = 0.0
        elif vol.vix_change < 2: scores["vix_velocity"] = 30.0
        elif vol.vix_change < 5: scores["vix_velocity"] = 70.0
        else: scores["vix_velocity"] = 100.0
            
        # 3. VIX Term Structure
        if vol.vix_term_structure == "contango": scores["term_structure"] = 10.0
        elif vol.vix_term_structure == "flat": scores["term_structure"] = 50.0
        elif vol.vix_term_structure == "backwardation": scores["term_structure"] = 90.0
        else: scores["term_structure"] = 50.0
            
        # 4. IV Rank for SPY
        scores["iv_rank"] = min(max(vol.iv_rank_spy, 0.0), 100.0)
        
        # 5. Put/Call Ratio (Extracted from OptionsSnapshot as requested)
        pcr = vol.put_call_ratio
        if "SPY" in vol.options:
            pcr = vol.options["SPY"].put_call_ratio
            
        if pcr < 0.8: scores["put_call_ratio"] = 10.0
        elif pcr < 1.0: scores["put_call_ratio"] = 30.0
        elif pcr < 1.2: scores["put_call_ratio"] = 50.0
        elif pcr < 1.5: scores["put_call_ratio"] = 80.0
        else: scores["put_call_ratio"] = 100.0
            
        # 6. Vol Risk Premium (Implied vs Realized)
        iv = 0.15
        if "SPY" in vol.options:
            iv = vol.options["SPY"].atm_iv
            
        rv = snap.macro.get("spy_realized_vol", iv)
        vrp = iv - rv
        
        if vrp < -0.05: scores["vol_risk_premium"] = 90.0
        elif vrp < 0.0: scores["vol_risk_premium"] = 70.0
        elif vrp < 0.05: scores["vol_risk_premium"] = 40.0
        else: scores["vol_risk_premium"] = 10.0
            
        # 7. Credit Spread Widths (Proxy via HYG and IEF)
        hyg = snap.quotes.get("HYG")
        ief = snap.quotes.get("IEF")
        if hyg and ief:
            # If IEF (Treasury) is up and HYG (Junk) is down, spreads are widening
            spread_proxy = ief.change_pct - hyg.change_pct
            if spread_proxy > 1.0: scores["credit_spread"] = 90.0
            elif spread_proxy > 0.5: scores["credit_spread"] = 70.0
            elif spread_proxy > 0.0: scores["credit_spread"] = 40.0
            else: scores["credit_spread"] = 10.0
        else:
            scores["credit_spread"] = 50.0
            
        return scores

    def _score_to_regime(self, score: float) -> str:
        if score < 25: return "low_vol"
        if score < 45: return "normal"
        if score < 65: return "elevated"
        if score < 85: return "high_vol"
        return "crisis"
        
    def _generate_parameters(self, regime: str) -> RegimeParameters:
        if regime == "low_vol":
            return RegimeParameters(
                regime_scale=1.2, min_confidence=0.50, max_open_decisions=12,
                iv_hv_zscore_threshold=1.5, max_position_pct=0.08, max_total_exposure_pct=0.40,
                max_daily_loss_pct=0.04, warning_drawdown_pct=0.10
            )
        elif regime == "normal":
            return RegimeParameters(
                regime_scale=1.0, min_confidence=0.55, max_open_decisions=10,
                iv_hv_zscore_threshold=2.0, max_position_pct=0.05, max_total_exposure_pct=0.30,
                max_daily_loss_pct=0.03, warning_drawdown_pct=0.08
            )
        elif regime == "elevated":
            return RegimeParameters(
                regime_scale=0.8, min_confidence=0.60, max_open_decisions=8,
                iv_hv_zscore_threshold=2.5, max_position_pct=0.04, max_total_exposure_pct=0.20,
                max_daily_loss_pct=0.02, warning_drawdown_pct=0.06
            )
        elif regime == "high_vol":
            return RegimeParameters(
                regime_scale=0.5, min_confidence=0.70, max_open_decisions=5,
                iv_hv_zscore_threshold=3.0, max_position_pct=0.02, max_total_exposure_pct=0.10,
                max_daily_loss_pct=0.015, warning_drawdown_pct=0.04
            )
        else: # crisis
            return RegimeParameters(
                regime_scale=0.1, min_confidence=0.85, max_open_decisions=2,
                iv_hv_zscore_threshold=4.0, max_position_pct=0.01, max_total_exposure_pct=0.05,
                max_daily_loss_pct=0.01, warning_drawdown_pct=0.02
            )

    def regime_distribution(self, lookback_days: int) -> Dict[str, float]:
        """Percentage of time in each regime over the lookback period (in snapshots)."""
        history_to_use = self.result_history[-lookback_days:] if lookback_days > 0 else self.result_history
        if not history_to_use:
            return {r: 0.0 for r in self.REGIMES}
            
        counts = {r: 0 for r in self.REGIMES}
        for res in history_to_use:
            counts[res.regime] += 1
            
        total = len(history_to_use)
        return {r: count / total for r, count in counts.items()}
        
    def avg_regime_duration(self) -> Dict[str, float]:
        """Average duration of each regime in snapshots."""
        if not self.result_history:
            return {r: 0.0 for r in self.REGIMES}
            
        durations = {r: [] for r in self.REGIMES}
        current_r = self.result_history[0].regime
        current_dur = 1
        
        for res in self.result_history[1:]:
            if res.regime == current_r:
                current_dur += 1
            else:
                durations[current_r].append(current_dur)
                current_r = res.regime
                current_dur = 1
        durations[current_r].append(current_dur)
        
        return {r: float(np.mean(durs)) if durs else 0.0 for r, durs in durations.items()}
        
    def current_regime_age(self) -> int:
        """How many snapshots the current regime has persisted."""
        if not self.result_history:
            return 0
        age = 0
        current_r = self.result_history[-1].regime
        for res in reversed(self.result_history):
            if res.regime == current_r:
                age += 1
            else:
                break
        return age
        
    def transition_matrix(self) -> Dict[str, Dict[str, float]]:
        """Probability matrix of regime-to-regime transitions."""
        matrix = {r1: {r2: 0.0 for r2 in self.REGIMES} for r1 in self.REGIMES}
        if len(self.result_history) < 2:
            return matrix
            
        transitions = {r: 0 for r in self.REGIMES}
        
        for i in range(1, len(self.result_history)):
            prev = self.result_history[i-1].regime
            curr = self.result_history[i].regime
            matrix[prev][curr] += 1
            transitions[prev] += 1
            
        for r1 in self.REGIMES:
            total = transitions[r1]
            if total > 0:
                for r2 in self.REGIMES:
                    matrix[r1][r2] /= total
                    
        return matrix
```

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, List, Optional

from command.autopilot.market_monitor import MarketSnapshot

# Safely import Signal and Direction to prevent ModuleNotFoundError in restricted environments
try:
    from command.autopilot.signal_collector import Signal, Direction
except ImportError:
    class Direction(str, Enum):
        LONG = "LONG"
        SHORT = "SHORT"
        HOLD = "HOLD"
    @dataclass
    class Signal:
        symbol: str
        direction: Direction
        confidence: float
        source: str
        entry_price: float = 0.0
        stop_loss: float = 0.0
        take_profit: float = 0.0
        setup_type: str = ""
        timeframe: str = "swing"

        timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))
        @property

class Action(str, Enum):
    OPEN_LONG = "OPEN_LONG"
    OPEN_SHORT = "OPEN_SHORT"
    CLOSE = "CLOSE"
    HEDGE = "HEDGE"
    REDUCE = "REDUCE"
    HOLD = "HOLD"
    SKIP = "SKIP"


@dataclass
class Decision:
    symbol: str
    action: Action
    confidence: float
    size_pct: float
    signals_used: int
    reasoning: str
    target_strategy: str = ""
    entry_price: float = 0.0
    stop_loss: float = 0.0
    take_profit: float = 0.0
    urgency: str = "normal"
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


@dataclass
class DecisionConfig:
    min_confidence: float = 0.55
    min_signals: int = 1
    max_position_pct: float = 0.10
    base_position_pct: float = 0.03
    kelly_fraction: float = 0.25
    max_open_decisions: int = 10
    # Updated to match the exact 5 new regime string values
    regime_scale: dict[str, float] = field(default_factory=lambda: {
        "low_vol": 1.2,
        "normal": 1.0,
        "elevated": 0.8,
        "high_vol": 0.5,
        "crisis": 0.1,
    })


class DecisionEngine:
    def __init__(self, config: DecisionConfig | None = None):
        self.config = config or DecisionConfig()
        
    def decide(self, signals: list[Signal], market: MarketSnapshot) -> list[Decision]:
        """Evaluates signals and sizes positions based on market regime."""
        
        # Default config values
        regime_scale = self.config.regime_scale.get(market.regime, 1.0)
        min_confidence = self.config.min_confidence
        max_open_decisions = self.config.max_open_decisions
        
        # Apply dynamic overrides if RegimeResult is attached to snapshot
        if hasattr(market, 'regime_result') and market.regime_result and market.regime_result.parameters:
            params = market.regime_result.parameters
            regime_scale = params.regime_scale
            min_confidence = params.min_confidence
            max_open_decisions = params.max_open_decisions
            
        decisions = []
        actionable_signals = [s for s in signals if s.is_actionable and s.confidence >= min_confidence]
        actionable_signals.sort(key=lambda s: s.confidence, reverse=True)
        actionable_signals = actionable_signals[:max_open_decisions]
        
        for sig in actionable_signals:
            action = Action.OPEN_LONG if sig.direction == Direction.LONG else Action.OPEN_SHORT
            
            # Position sizing scaled dynamically by regime
            size_pct = self.config.base_position_pct * regime_scale * (sig.confidence / min_confidence)
            size_pct = min(size_pct, self.config.max_position_pct)
            
            decisions.append(Decision(
                symbol=sig.symbol,
                action=action,
                confidence=sig.confidence,
                size_pct=size_pct,
                signals_used=1,
                reasoning=f"Signal confidence {sig.confidence:.2f} >= {min_confidence:.2f} in {market.regime} regime",
                entry_price=sig.entry_price,
                stop_loss=sig.stop_loss,
                take_profit=sig.take_profit
            ))
            
        return decisions

    assert res3.transition_event.from_regime == "normal"
